function calcPrivateMinutes(privateTimes, tripStart, tripEnd) {

  // Calculate total private minutes that fall within trip period

  const ts = new Date(tripStart).getTime();

  const te = new Date(tripEnd).getTime();

  let totalMinutes = 0;

  for (const pt of privateTimes) {

    let ps = new Date(pt.start).getTime();

    let pe = new Date(pt.end).getTime();

    // Clamp to trip boundaries

    ps = Math.max(ps, ts);

    pe = Math.min(pe, te);

    if (pe > ps) {

      totalMinutes += (pe - ps) / 60000;

    }

  }

  return totalMinutes;

}


function formatMinutes(mins) {

  const d = Math.floor(mins / 1440);

  const h = Math.floor((mins % 1440) / 60);

  const m = Math.round(mins % 60);

  let s = '';

  if (d) s += `${d}d `;

  if (h || d) s += `${h}h `;

  s += `${m}min`;

  return s;

}


async function calcDnevnice() {

  const start = document.getElementById('trip_start_datetime').value;

  const end = document.getElementById('trip_end_datetime').value;

  if (!start || !end) return;


  const privateTimes = getPrivateTimes();

  const privateMinutes = calcPrivateMinutes(privateTimes, start, end);


  const res = await fetch('/api/calculate_dnevnice', {

    method: 'POST',

    headers: {'Content-Type': 'application/json'},

    body: JSON.stringify({ start, end, private_minutes: privateMinutes })

  });

  const data = await res.json();

  dnevniceData = { count: data.dnevnice, rate: data.total / (data.dnevnice || 1) || 0, total: data.total };

  // Update hidden fields

  document.getElementById('daily_allowance_count').value = data.dnevnice;

  document.getElementById('daily_allowance_rate').value = dnevniceData.rate;

  document.getElementById('daily_allowance_total').value = data.total;

  // Update display

  const box = document.getElementById('dnevnice-box');

  box.style.display = 'flex';


  // Total trip duration

  const totalMs = new Date(end) - new Date(start);

  const totalMinutes = totalMs / 60000;

  const totalDays = Math.floor(totalMinutes / 1440);

  const totalHours = Math.floor((totalMinutes % 1440) / 60);

  const totalMins = Math.round(totalMinutes % 60);

  document.getElementById('calc-duration').textContent = `${totalDays}d ${totalHours}h ${totalMins}min`;


  // Private time display

  const privateWrap = document.getElementById('calc-private-wrap');

  const officialWrap = document.getElementById('calc-official-wrap');

  if (privateMinutes > 0) {

    privateWrap.style.display = '';

    officialWrap.style.display = '';

    document.getElementById('calc-private').textContent = `− ${formatMinutes(privateMinutes)}`;

    const officialMinutes = totalMinutes - privateMinutes;

    document.getElementById('calc-official').textContent = formatMinutes(Math.max(0, officialMinutes));

  } else {

    privateWrap.style.display = 'none';

    officialWrap.style.display = 'none';

  }


  // Dnevnice from API response (API should subtract private time)

  document.getElementById('calc-count').textContent = data.dnevnice;

  document.getElementById('calc-rate').textContent = `${dnevniceData.rate.toFixed(2)} €`;

  document.getElementById('calc-total').textContent = `${data.total.toFixed(2)} €`;

  updateTotals();

}

async function autoFillKmFromVehicleLog() {

  const orderId = document.getElementById('order-id')?.value;

  if (!orderId) return; // novi nalog — nema što dohvaćati


  // Provjeri jesu li km polja već popunjena

  const startKmEl = document.getElementById('start_km');

  const endKmEl   = document.getElementById('end_km');

  if (!startKmEl || !endKmEl) return;


  const hasStartKm = startKmEl.value && parseFloat(startKmEl.value) > 0;

  const hasEndKm   = endKmEl.value   && parseFloat(endKmEl.value) > 0;


  // Ako su oba već popunjena, ne diraj

  if (hasStartKm && hasEndKm) return;


  try {

    const res = await fetch(`/api/orders/${orderId}/km-from-log`);

    if (!res.ok) return;

    const d = await res.json();


    if (!d.found) return;


    let filled = [];


    if (!hasStartKm && d.start_km) {

      startKmEl.value = Math.round(d.start_km);

      filled.push('početna KM');

    }

    if (!hasEndKm && d.end_km) {

      endKmEl.value = Math.round(d.end_km);

      filled.push('završna KM');

    }


    if (filled.length > 0) {

      // Prikaži info badge ispod vozilo sekcije

      const vehicleSection = startKmEl.closest('.card');

      if (vehicleSection && !document.getElementById('km-autofill-badge')) {

        const badge = document.createElement('div');

        badge.id = 'km-autofill-badge';

        badge.style.cssText = 'background:#e8f4fd;border:1px solid #aac4db;border-radius:6px;padding:7px 12px;margin-top:8px;font-size:12px;color:#1a6b9a;display:flex;align-items:center;gap:6px;';

      }

    }

  } catch(e) {

  }

}
